    function fuzzyFind(str, list) {
  const reg = new RegExp('(.*)(' + str.split('').join(')(.*)(') + ')(.*)');
  let ans = [];
  list.forEach((ele) => {
    if (reg.test(ele)) ans.push(ele);
  });
  return ans;
}
