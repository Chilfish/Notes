    # 周期循环

    • 题目链接open in new window


    • AC 码

      #include <bits/stdc++.h>
using namespace std;
typedef long long ll;

int main()
{
    ll n, quan, yu;
    ll a[5] = {0};   //实际上余数就只有0 1 4 三种情况
    ll num[5] = {-1, 1, 4, 1, 0};  //余数的周期是4 ： 1 4 1 0

    cin >> n;
    quan = n / 4, yu = n % 4;
    a[0] = quan, a[1] = quan * 2, a[4] = quan;

    for (ll i = 1; i <= yu; i++) //周期之外
        a[num[i]]++;

    ll hv = (a[0] != 0) + (a[1] != 0) + (a[4] != 0);
    cout << hv << "\n";

    for (ll i = 0; i <= 4; i++)
        if (a[i] != 0)
            cout << i << " " << a[i] << "\n";
}

    # 拼接数字取模

    • 题目链接open in new window


    • AC 码

      int n, t;
signed main()
{
    for (cin >> t; t--;)
    {
        cin >> n;
        int x = 0, ans = 0;
        for (int i = 1; i <= n; i++)
        {
            x = stoi(to_string(x) + to_string(i)) % 7;
            ans += !x;
        }
        cout << ans << endl;
    }
}

    # 幂运算 快速幂思想

    • 题目链接open in new window


    • 题解

        观察到，将序列 BBB 的 下标 + 1 以 二进制 列出后：

      0001001000110100010101100111
      1349101213

      3 的幂次数的下标为 0001， 0010， 0100， 1000   不难得出：类比二进制或快速幂的方法，将幂次数相加即为 二进制数为 1 的幂数相加

      • 快速幂:
        8421
      • 幂次和：
        \k3\k^3\k3k2k^2k2k1k^1k1k0k^0k0

        例如：当 n=5,k=3n = 5, k = 3n=5,k=3 时， nnn 的 二进制 为 010101010101 ，即为 32+30=103^2 + 3^0 = 1032+30=10


    • AC 码

      const int mod = 1e9 + 7;
int main()
{
    ll n, k, ans = 0, t = 1;
    cin >> n >> k;
    while (n)
    {
        if (n & 1)  (ans += t) %= mod;
        (t *= k) %= mod, n >>= 1;
    }
    cout << ans;
}
