    # 快速幂

    比一般的 pow 快，且 x mod y=x,y>xx \bmod y = x , y > xxmody=x,y>x

    ll fpow(ll a, ll x, ll mod) {
  ll ans = 1;
  while (x) {
    if (x & 1) (ans *= a) %= mod;
    (a *= a) %= mod, x >>= 1;
  }
  return ans % mod;
}

    # 快速乘

    利用了 unsigned long long 的自动溢出，且保证了它们溢出后的差值基本不变。且 O(1)O(1)O(1) ！

    ll ksc(ll x, ll y, ll mod) {
  ll z = (long double)x / mod * y;
  ll res = (ull)x * y - (ull)z * mod;
  return (res + mod) % mod;
}

    # 最大公约数

    最小公倍数为 a * b / 返回值 a。在 C++中：c = __gcd(a, b);

    int gcd(int a, int b) {
  while (a != b)
    (a > b) ? (a -= b) : (b -= a);
  return a;
}

    且：辗转相减法求 GCD 的次数 竟然是 辗转相除法 每一次操作的 modmodmod 的相加

    # 完全平方数

    原理：
    • 观察到：1=1,4=1+3,  9=1+3+5,  16=1+3+5+71=1, 4=1+3 ,\; 9=1+3+5 ,\; 16=1+3+5+71=1,4=1+3,9=1+3+5,16=1+3+5+7
    • 以此类推，可以从 nnn 开始 不断减去一个从 1 开始不断增大的奇数，若最终减成了 0，说明是完全平方数，否则不是

    字面解： O(n)O(\sqrt{n})O(n​)

    bool PerfectSquare(ll n) {
  for (ll t = 1; n > 0; t += 2)
    n -= t;
  return !n;
}

    优化后： O(n4)O(\sqrt[4]{n})O(4n​) ，最坏也有 O(n)O(\sqrt{n})O(n​)

    bool Perfect(ll n) {
  for (ll i = 2; i <= n / i; ++i)
    while (n % (i * i) == 0)
      n /= i * i;
  return n == 1;
}

    bool isprime(ll n) {
  if (n < 2 or n & 1 == 0)
    return false;
  for (ll i = 3; i <= n / i; i += 2)
    if (n % i == 0)
      return false;
  return true;

    倍数筛：

    int a[10000];
int primer(int n) {
  bool p[n] = {false}; //本来全是true的
  int k = 0;
  for (int i = 2; i <= n; i++)
    if (!p[i]) {
      a[k++] = i;
      for (int j = i * i; j <= n; j += i)
        p[j] = true;  //其实为了方便反了过来
    }
  return k;  //k为最后一个质数的位置

    欧拉筛

    const int N = 1e7 + 5;
bool f[N]{}; // 判断 n 是否为质数
int primes[N]{}; // 返回第 n 个质数

void ola() {
  memset(f, true, N);
  for (int i = 2, cnt = 0; i <= N; i++) {
    if (f[i]) primes[cnt++] = i;
    for (int j = 0; primes[j] <= N / i; j++) {
      f[primes[j] * i] = false;
      if (i % primes[j] == 0) break;
    }
  }
}

vi add(vi x, vi y) {
  if (x.size() < y.size()) return add(y, x);
  vi ans;
  int t = 0;
  for (int i = 0; i < x.size(); ++i) {

  if (t)

}

    ll C(int n, int m) {
  ll ans = 1;
  n = min(n, m - n);
  for (int i = m, j = n; j; i--, j--)

    # 蔡勒日期公式

    用于判断某年的某一日是星期几

    公式： week=(y+[y4]+[c4]−2c+[26×(m+1)10]+d−1) mod 7week=(y+ \left [ \frac{y}{4}\right ] + \left [ \frac{c}{4} \right ] - 2c + \left[\frac{26 \times (m+1)}{10} \right ] + d -1) \bmod 7week=(y+[4y​]+[4c​]−2c+[1026×(m+1)​]+d−1)mod7

    • weekweekweek：以周日为第一天
    • ccc：世纪
    • [x]\left [ x \right][x] ：对 xxx 取整

    且：在 蔡勒公式 中，某年的 1、2 月要看作 上一年 的 13、14 月来计算。比如 2003 年 1 月 1 日要看作 2002 年的 13 月 1 日来计算

    string week[7] = {"Sun", "Mon", "Tue", "Wed", "Thur", "Fri", "Sat"};
string DayOfTheWeek(int year, int month, int day) {
  if (month <= 2)
    month += 12, year -= 1;
  int c = year / 100;
  year %= 100;
  int w = (year + year / 4 + c / 4 - 2 * c + (26 * (month + 1) / 10) + day - 1) % 7;
  return week[(w + 7) % 7];
}

    # Miller-Robin 的素数判断

    利用随机化算法判断一个数是合数还是可能是素数（毫秒级判断大数）。（玄学）

    bool Miller_Rabbin(int x) {
  if (x == 1) return false;
  if (x == 2) return true;
  for (int i = 1; i <= 30; ++i) {
    int now = rand() % (x - 2) + 2;
    if (int(fpow(now, x - 1, x)) != 1)
      return false;
  }
  return true;
}
